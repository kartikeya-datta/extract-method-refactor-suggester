        if (!jars.contains(jarFileName)) {
            Constants.message("jsp.message.adding_jar",
                              new Object[] { jarFileName },
                              Log.INFORMATION);
            
            jars.addElement(jarFileName);
        }
    }

        StringBuffer cpath = new StringBuffer();

        }
