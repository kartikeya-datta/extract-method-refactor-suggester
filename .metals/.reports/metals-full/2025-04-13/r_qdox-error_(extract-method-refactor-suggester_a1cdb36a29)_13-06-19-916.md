	private HttpEntity() {
		this(null, null);
	}

	/**

	public HttpEntity(T body) {
		this(body, null);

	 * Returns the body of this entity.

	public T getBody() {
		return this.body;
	}

	/**
	 * Indicates whether this entity has a body.
	 */

	}
